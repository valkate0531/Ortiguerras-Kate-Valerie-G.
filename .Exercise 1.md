.Exercise 1



Show product names with supplier names



SELECT product\_name, supplier\_name

FROM products

JOIN suppliers USING (supplier\_id);



.Exercise 2



Show each sale with product name, quantity, and date



SELECT product\_name, quantity, sale\_date

FROM sales

JOIN products USING (product\_id);



.Exercise 3



Products from “TechSource Inc.”



SELECT product\_name

FROM products

JOIN suppliers USING (supplier\_id)

WHERE supplier\_name = 'TechSource Inc.';



.Exercise 4



Total quantity sold per product



SELECT product\_name, SUM(quantity)

FROM sales

JOIN products USING (product\_id)

GROUP BY product\_name;



.Exercise 5



Total sales amount per product



SELECT product\_name, SUM(quantity \* price)

FROM sales

JOIN products USING (product\_id)

GROUP BY product\_name;



.Exercise 6



Total revenue per supplier



SELECT supplier\_name, SUM(quantity \* price)

FROM sales

JOIN products USING (product\_id)

JOIN suppliers USING (supplier\_id)

GROUP BY supplier\_name;



.Exercise 7



Products with remaining stock



SELECT product\_name, (stock\_in - stock\_out) AS remaining\_stock

FROM products

JOIN inventory USING (product\_id);



.Exercise 8



Products sold at least 5 times



SELECT product\_name, SUM(quantity)

FROM sales

JOIN products USING (product\_id)

GROUP BY product\_name

HAVING SUM(quantity) >= 5;



.Exercise 9



Most sold product



SELECT product\_name, SUM(quantity)

FROM sales

JOIN products USING (product\_id)

GROUP BY product\_name

ORDER BY SUM(quantity) DESC

LIMIT 1;



.Exercise 10



Total sales per category



SELECT category, SUM(quantity \* price)

FROM sales

JOIN products USING (product\_id)

GROUP BY category;
